下载进度

axios.create({ // 下载文件
    baseURL: process.env.VUE_APP_EPUB_URL,
    method: 'get',
    responseType: 'blob',
    timeout: 180 * 1000, // 3分钟超时
    onDownloadProgress: progressEvent => { // axios监听下载进度
        if (onProgress) onProgress(progressEvent)
    }
}).get(`www/abc.xlsx`)
